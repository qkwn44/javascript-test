{
    function func(){
        let i = 5, j = 4, k = 1, l,m;
        l = i > 5 || j != 0;
        m = j <= 4 && k < 1;

        document.write(m);
    }
    func();

    l은 i의 값이 5보다 크거나, j의 값이 0이 아니거나

    답 : l은 둘 중에 하나의 조건만 성립하면 되기 때문에 true
         m은 두 조건 모두 성립해야하기 때문에 false

02. 할당 연산자 

{
    function func(){
        let i = 10, j= 10, k=30;
        i /= j;
        j -=i;
        k %= j;
        document.write(i);
        document.write(j);
        document.write(k);
    }
    func();
    1. i /= j 나누기 할당 연산자 
        i와 j의 값이 각각 10이므로 i = 1
    2. j -= i 빼기 할당 연산자
        i = 1, j = 10 이므로 , j = 9
    3. k %= j; 나누기 할당 연산자
        나눈 나머지 값 출력
        k = 3

05. 더하기 할당 연산자 += 

{
    function func(){
        let a = [];
        for( i=1; i <=5; i++){
            a +=[i];
        }
        document.write(a);
    }
    func();

    + 정답 +
    12345
    ---------------------------------------
    // foo = 'foo', bar = 5 , baz = true
// 위와 같은 변수를 가정할 때
// Number + Number -> 덧셈
bar += 2 // 7
// Boolean + Number -> 덧셈
baz += 1 // 2
// Boolean + Boolean -> 덧셈
baz += false // 1
// Number + String -> 연결
bar += 'foo' // "5foo"
// String + Boolean -> 연결
foo += false // "foofalse"
// String + String -> 연결
foo += 'bar' // "foobar"

06. 데이터값 불러오기와 반복문 

{
    function func(){
        let a = [1,2,3,4,5,6,7,8,9], cnt = 0;
        for( i=0; i< a.length; i++ ){
            if(a[i]%2 != 0){
                cnt = cnt +1;
            }
        }
        document.write(cnt);
    }
    func();
    
    ++ 풀이 ++
    i값 증감하는 반복문을 i < 9 일때 까지 반복 해라
    i = 0 일 때 a[i]=1이기 때문에 cnt =1 
    i = 1 일 때 a[i]=2이기 때문에 나머지가 0이므로 cnt 값 출력 하지 않음
    i = 2 일 때 a[i]=3 , cnt = 2
    i = 3     , a[i]=4 , 값 출력 x
    ...
    i = 8     , a[i]=9 , cnt = 5

    정답 document.write(cnt) = 5 

    ++ for문 ++
    for ([초기문]; [조건문]; [증감문]){
        반복할 코드
    }
    초기문 실행 후 반복할 코드 실행, 참일 경우 증감문 실행 후 다시 반복코드 실행

}
